## Props

<PropsTable 
rows={[
    {name:"header",type:"string",defaultValue:"",required:false,description:"The text that will be displayed in the header of the card"},
    {name:"children",type:"React.ReactNode",defaultValue:"",required:true,description:"The content that will be displayed in the body of the card"},
    {name:"logo",type:"'none' | 'default' | 'document'",defaultValue:"'default'",required:false,description:"Choose the logo in the corner"},
    {name:"elevation",type:"a number between 1 and 5",defaultValue:"1",required:false,description:"Defined the elevation of the card"},
    {name:"customLogo",type:"React.ReactNode",defaultValue:"",required:false,description:"The logo that wil be displayed in the corner instead of the defaults logos"},
    {name:"logoSize",type:"'medium' | 'small'",defaultValue:"'medium'",required:false,description:"Defined the size of the logo in the corner"},
    {name:"className",type:"string",defaultValue:"",required:false,description:"Add the className to the card and override the default properties"},
    {name:"style",type:"React.CSSProperties",defaultValue:"",required:false,description:"Add the style to the card and override the default properties"},
]} />
